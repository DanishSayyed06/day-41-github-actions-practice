#Cron

name: pr-check

on:
  pull_request:
    types: [opened, synchronize, reopened]
    branches:
      - main
  workflow_dispatch:
  schedule:
    - cron: '0 0 * * *'

jobs:
  check-pr:
    runs-on: ubuntu-latest
    steps:
      - name: Print branch name
        run: |
          echo "PR check running for branch: ${{ github.head_ref }}"
